## üìä BACKLOG GROOMING

```python
async def conduct_backlog_grooming(project_id: str):
    """
    –ü—Ä–æ–≤–µ—Å—Ç–∏ grooming session –¥–ª—è backlog.

    –¶–µ–ª–∏:
    - –ü—Ä–∏–æ—Ä–∏—Ç–∏–∑–∏—Ä–æ–≤–∞—Ç—å –∑–∞–¥–∞—á–∏
    - –û—Ü–µ–Ω–∏—Ç—å —Å–ª–æ–∂–Ω–æ—Å—Ç—å
    - –†–∞–∑–±–∏—Ç—å –±–æ–ª—å—à–∏–µ –∑–∞–¥–∞—á–∏
    - –£–¥–∞–ª–∏—Ç—å —É—Å—Ç–∞—Ä–µ–≤—à–∏–µ
    """

    print("üìä BACKLOG GROOMING")
    print("‚îÅ" * 50)

    # –ü–æ–ª—É—á–∏—Ç—å –≤–µ—Å—å backlog
    backlog = await mcp__archon__find_tasks(
        project_id=project_id,
        filter_by="status",
        filter_value="todo"
    )

    grooming_results = {
        "prioritized": 0,
        "split": 0,
        "archived": 0,
        "clarified": 0
    }

    for task in backlog:
        # 1. –ü—Ä–æ–≤–µ—Ä–∏—Ç—å –∞–∫—Ç—É–∞–ª—å–Ω–æ—Å—Ç—å
        if is_outdated(task):
            await archive_task(task["id"])
            grooming_results["archived"] += 1
            continue

        # 2. –ü—Ä–æ–≤–µ—Ä–∏—Ç—å –æ–ø–∏—Å–∞–Ω–∏–µ
        if needs_clarification(task):
            await clarify_task_requirements(task["id"])
            grooming_results["clarified"] += 1

        # 3. –ü—Ä–æ–≤–µ—Ä–∏—Ç—å —Ä–∞–∑–º–µ—Ä
        if is_too_large(task):
            subtasks = split_task(task)
            for subtask in subtasks:
                await mcp__archon__manage_task(
                    action="create",
                    project_id=project_id,
                    **subtask
                )
            grooming_results["split"] += 1

        # 4. –ü—Ä–∏–æ—Ä–∏—Ç–∏–∑–∏—Ä–æ–≤–∞—Ç—å
        await mcp__archon__manage_task(
            action="update",
            task_id=task["id"],
            task_order=calculate_priority(task)
        )
        grooming_results["prioritized"] += 1

    print(f"\n‚úÖ Grooming –∑–∞–≤–µ—Ä—à–µ–Ω:")
    print(f"   –ü—Ä–∏–æ—Ä–∏—Ç–∏–∑–∏—Ä–æ–≤–∞–Ω–æ: {grooming_results['prioritized']}")
    print(f"   –†–∞–∑–±–∏—Ç–æ: {grooming_results['split']}")
    print(f"   –ê—Ä—Ö–∏–≤–∏—Ä–æ–≤–∞–Ω–æ: {grooming_results['archived']}")
    print(f"   –£—Ç–æ—á–Ω–µ–Ω–æ: {grooming_results['clarified']}")

    return grooming_results
```

```python
    for keep in keeps:
        print(f"   ‚Ä¢ {keep}")

    # –ß—Ç–æ –Ω–µ —Ö–æ—Ä–æ—à–æ (Problems)

    print(f"\n‚ö†Ô∏è –ü–†–û–ë–õ–ï–ú–´:")
    for problem in problems:
        print(f"   ‚Ä¢ {problem}")

    # –ß—Ç–æ —É–ª—É—á—à–∏—Ç—å (Actions)
    actions = generate_action_items(problems)
    print(f"\nüéØ ACTION ITEMS:")
    for i, action in enumerate(actions, 1):
        print(f"   {i}. {action}")
        # –°–æ–∑–¥–∞—Ç—å –∑–∞–¥–∞—á–∏ –¥–ª—è action items
        await mcp__archon__manage_task(
            action="create",
            project_id=project_id,
            title=f"[PROCESS] {action}",
            assignee="Archon Project Manager",
            task_order=90  # High priority
        )

    retrospective = {
        "sprint": sprint_number,
        "completion_rate": completion_rate,
        "velocity": calculate_velocity(completed),
        "keeps": keeps,
        "problems": problems,
        "actions": actions
    }

    return retrospective
```

## üìà VELOCITY TRACKING

### –û—Ç—Å–ª–µ–∂–∏–≤–∞–Ω–∏–µ –ø—Ä–æ–¥—É–∫—Ç–∏–≤–Ω–æ—Å—Ç–∏ –∫–æ–º–∞–Ω–¥—ã

```python
def calculate_team_velocity(project_id: str, sprints_count: int = 3):
    """
    –í—ã—á–∏—Å–ª–∏—Ç—å —Å—Ä–µ–¥–Ω—é—é velocity –∫–æ–º–∞–Ω–¥—ã –∑–∞ –ø–æ—Å–ª–µ–¥–Ω–∏–µ —Å–ø—Ä–∏–Ω—Ç—ã.

    Args:
        project_id: UUID –ø—Ä–æ–µ–∫—Ç–∞
        sprints_count: –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å–ø—Ä–∏–Ω—Ç–æ–≤ –¥–ª—è –∞–Ω–∞–ª–∏–∑–∞

    Returns:
        float: –°—Ä–µ–¥–Ω—è—è velocity (story points per sprint)
    """

    velocities = []

    for sprint in range(1, sprints_count + 1):
        sprint_tasks = get_sprint_tasks(project_id, sprint)
        completed = [t for t in sprint_tasks if t["status"] == "done"]
        velocity = sum(t.get("story_points", 1) for t in completed)
        velocities.append(velocity)

    average_velocity = sum(velocities) / len(velocities)

    print(f"üìà VELOCITY ANALYSIS:")
    print(f"   Last {sprints_count} sprints: {velocities}")
    print(f"   Average: {average_velocity:.1f} story points/sprint")

    return average_velocity
```
