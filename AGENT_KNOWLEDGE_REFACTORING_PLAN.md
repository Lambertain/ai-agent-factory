import re
from pathlib import Path
from typing import Tuple

def extract_sections(content: str) -> Tuple[str, str, str]:
    """
    –ò–∑–≤–ª–µ—á—å —Å–µ–∫—Ü–∏–∏ –∏–∑ knowledge —Ñ–∞–π–ª–∞.

        (header, system_prompt, domain_knowledge)

    header_pattern = r"^(# ‚ö†Ô∏è.*?^---\n\n)"
    header_match = re.search(header_pattern, content, re.MULTILINE | re.DOTALL)

    # –ò–∑–≤–ª–µ—á—å —Å–∏—Å—Ç–µ–º–Ω—ã–π –ø—Ä–æ–º–ø—Ç (–ø–æ—Å–ª–µ —Ö–µ–¥–µ—Ä–∞, ~50 —Å—Ç—Ä–æ–∫)
    prompt_pattern = r"^## üé≠ –°–ò–°–¢–ï–ú–ù–´–ô –ü–†–û–ú–ü–¢.*?(?=^##|\Z)"
    prompt_match = re.search(prompt_pattern, content, re.MULTILINE | re.DOTALL)

    # –û—Å—Ç–∞–ª—å–Ω–æ–µ = –¥–æ–º–µ–Ω–Ω—ã–µ –∑–Ω–∞–Ω–∏—è
    domain_start = prompt_match.end() if prompt_match else len(header_match.group(0))
    domain_knowledge = content[domain_start:].strip()

    return (
        header_match.group(0) if header_match else "",
        prompt_match.group(0) if prompt_match else "",
        domain_knowledge
    )

def create_new_knowledge(
    agent_name: str,
    system_prompt: str,
    domain_knowledge: str,
    template_path: Path
) -> str:
    """
    –°–æ–∑–¥–∞—Ç—å –Ω–æ–≤—ã–π knowledge —Ñ–∞–π–ª –Ω–∞ –æ—Å–Ω–æ–≤–µ —à–∞–±–ª–æ–Ω–∞.
    """
    template = template_path.read_text(encoding='utf-8')

    # –ó–∞–º–µ–Ω–∏—Ç—å –ø–ª–µ–π—Å—Ö–æ–ª–¥–µ—Ä—ã
    new_content = template.replace(
        "[–ù–ê–ó–í–ê–ù–ò–ï –†–û–õ–ò]", agent_name.replace("_", " ").title()

    # –í—Å—Ç–∞–≤–∏—Ç—å —Å–∏—Å—Ç–µ–º–Ω—ã–π –ø—Ä–æ–º–ø—Ç
    new_content = new_content.replace(
        "## üé≠ –°–ò–°–¢–ï–ú–ù–´–ô –ü–†–û–ú–ü–¢ –†–û–õ–ò: [–ù–ê–ó–í–ê–ù–ò–ï –†–û–õ–ò]",
        system_prompt
    )

    # –î–æ–±–∞–≤–∏—Ç—å –¥–æ–º–µ–Ω–Ω—ã–µ –∑–Ω–∞–Ω–∏—è –ø–æ—Å–ª–µ –ø—Ä–æ–º–ø—Ç–∞
    insert_point = new_content.find("## üîç –î–û–ú–ï–ù–ù–´–ï –ó–ù–ê–ù–ò–Ø")
    if insert_point != -1:
        new_content = (
            new_content[:insert_point] +
            domain_knowledge + "\n\n" +
            new_content[insert_point:]
        )

    return new_content

    –†–µ—Ñ–∞–∫—Ç–æ—Ä–∏—Ç—å –æ–¥–∏–Ω –∞–≥–µ–Ω—Ç.

    template = Path("agents/_shared/agent_knowledge_template.md")
